299 !-----------------------------------------------------------------------
300 !
301 ! Setup the material according to mat_opt.
302 !
303 ! There are only two materials max; one per cell. mat_opt defines the
304 ! material layout and has a similar meaning for 1-D, 2-D, 3-D problems.
305 !
306 ! 0 = Homogeneous (mat1) problem, regardless of dimension
307 ! 1 = Center. mat1 is the base. 1-D: 25%mat1/50%mat2/25%mat1
308 ! 2/D: same as 1-D but in two dimensions, so mat2 region is a square
309 ! 3/D: same as 1-D but in three dimensions, so mat2 is a cube
310 ! 2 = Corner. Same concept as 1, but move slab/square/cube to origin
311 !
312 ! Return starting/ending indices for printint an echo.
313 !
314 !-----------------------------------------------------------------------
315 
316  INTEGER(i_knd), INTENT(OUT) :: i1, i2, j1, j2, k1, k2
317 !_______________________________________________________________________
318 !
319 ! Local variables
320 !_______________________________________________________________________
321 
322  INTEGER(i_knd) :: i, j, k, jj, kk
323 !_______________________________________________________________________
324 !
325 ! Form the base with mat1. Use dimension and mat_opt to determine
326 ! the rest of the layout.
327 !_______________________________________________________________________
328 
329  mat = 1
330 
331  i1 = 1; i2 = 1
332  j1 = 1; j2 = 1
333  k1 = 1; k2 = 1
334  IF ( mat_opt == 1 ) THEN
335  i1 = nx/4 + 1
336  i2 = 3*nx/4
337  IF ( ndimen > 1 ) THEN
338  j1 = ny_gl/4 + 1
339  j2 = 3*ny_gl/4

341  k1 = nz_gl/4 + 1
342  k2 = 3*nz_gl/4
343  END IF
344  END IF
345  ELSE IF ( mat_opt == 2 ) THEN
346  i2 = nx/2
347  IF ( ndimen > 1 ) THEN
348  j2 = ny_gl/2
349  IF ( ndimen > 2 ) k2 = nz_gl/2
350  END IF
351  END IF
352 
353  IF ( mat_opt > 0 ) THEN
354  DO k = k1, k2
355  IF ( klb<=k .AND. k<=kub ) THEN
356  kk = mod( k-1, nz ) + 1
357  DO j = j1, j2
358  IF ( jlb<=j .AND. j<=jub ) THEN
359  jj = mod( j-1, ny ) + 1
360  DO i = i1, i2
361  mat(i,jj,kk) = 2
362  END DO
363  END IF
364  END DO

376 !-----------------------------------------------------------------------
377 !
378 ! Setup the material according to src_opt.
379 !
380 ! Source is either on at strength 1.0 or off, per cell. src_opt defines
381 ! the layout and has a similar meaning for 1-D, 2-D, 3-D problems.
382 !
383 ! 0 = Source everywhere, regardless of dimension
384 ! 1 = Source occupies the slab/square/cube center of the problem
385 ! 1/D: 25%/50%/25% = source off/on/off
386 ! 2/D: same as 1-D but in two dimensions
387 ! 3/D: same as 1-D but in three dimensions
388 ! 2 = Corner. Same concept as 1, but move slab/square/cube to origin
389 ! 3 = MMS option, f = sin(pi*x/lx)
390 ! All options isotropic source
391 !
392 ! Return starting/ending indices for printing an echo.
393 !
394 !-----------------------------------------------------------------------
395 
396  CHARACTER(LEN=64), INTENT(OUT) :: error

402 !_______________________________________________________________________
403 
404  INTEGER(i_knd) :: i, j, k, jj, kk
405 !_______________________________________________________________________
406 !

413  return

418 !_______________________________________________________________________

470  SUBROUTINE setup_data
471 
472 !-----------------------------------------------------------------------
473 !
474 ! Setup the data arrays according to simple rules/functions to scale
475 ! to arbitrary number of groups, moments. Fixed at a library for two
476 ! materials only. Fixed group-to-group coupling properties.
477 !
478 !-----------------------------------------------------------------------
479 !_______________________________________________________________________
480 !
481 ! Local variables
482 !_______________________________________________________________________
483 
484  INTEGER(i_knd) :: g, n
485 
486  REAL(r_knd) :: t
487 !_______________________________________________________________________
488 !
489 ! Set the group 1 data for material 1 and material 2 (if present).
490 !_______________________________________________________________________
491 
492  sigt(1,1) = one
493  siga(1,1) = half
494  sigs(1,1) = half
495 
496  IF ( nmat == 2 ) THEN
497  sigt(2,1) = two
498  siga(2,1) = 0.8_r_knd
499  sigs(2,1) = 1.2_r_knd
500  END IF
501 !_______________________________________________________________________
502 !
503 ! Scale these values for increasing number of groups by adding 0.01
504 ! to sigt for each additional group. Split that 0.01 evenly among
505 ! siga and sigs.
506 !_______________________________________________________________________
507 
508  DO g = 2, ng
509  sigt(:,g) = sigt(:,g-1) + 0.01_r_knd
510  siga(:,g) = siga(:,g-1) + 0.005_r_knd
511  sigs(:,g) = sigs(:,g-1) + 0.005_r_knd
512  END DO
513 !_______________________________________________________________________
514 !
515 ! For material 1, upscattering from a given group to all above groups
516 ! is 10% of total scattering. 20% is in group. 70% is down-scattering.
517 ! For group 1/ng, no up/down-scattering, so self-scattering takes the
518 ! remaining fraction.
519 !_______________________________________________________________________
520 
521  DO g = 1, ng
522 
523  IF ( ng == 1 ) THEN
524  slgg(1,1,1,1) = sigs(1,g)
525  exit
526  END IF
527 
528  slgg(1,1,g,g) = 0.2_r_knd*sigs(1,g)
529 
530  IF ( g > 1 ) THEN
531  t = one / REAL( g-1, r_knd )
532  slgg(1,1,g,1:(g-1)) = 0.1_r_knd*sigs(1,g)*t
533  ELSE
534  slgg(1,1,1,1) = 0.3_r_knd*sigs(1,1)
535  END IF
536 
537  IF ( g < ng ) THEN
538  t = one / REAL( ng-g, r_knd )
539  slgg(1,1,g,(g+1):ng) = 0.7_r_knd*sigs(1,g)*t
540  ELSE
541  slgg(1,1,ng,ng) = 0.9_r_knd*sigs(1,ng)
542  END IF
543 
544  END DO
545 
546  IF ( nmat == 2 ) THEN
547 !_______________________________________________________________________
548 !
549 ! Repeat for material 2. Up-scattering is 10%. In-scattering is 50%.
550 ! Down-scattering is 40%.
551 !_______________________________________________________________________
552 
553  DO g = 1, ng
554 
555  IF ( ng == 1 ) THEN
556  slgg(2,1,1,1) = sigs(2,g)
557  exit
558  END IF
559 
560  slgg(2,1,g,g) = 0.5_r_knd*sigs(2,g)
561 
562  IF ( g > 1 ) THEN
563  t = one / REAL( g-1, r_knd )
564  slgg(2,1,g,1:(g-1)) = 0.1_r_knd*sigs(2,g)*t
565  ELSE
566  slgg(2,1,1,1) = 0.6_r_knd*sigs(2,1)
567  END IF
568 
569  IF ( g < ng ) THEN
570  t = one / REAL( ng-g, r_knd )
571  slgg(2,1,g,(g+1):ng) = 0.4_r_knd*sigs(2,g)*t
572  ELSE
573  slgg(2,1,ng,ng) = 0.9_r_knd*sigs(2,ng)
574  END IF
575 
576  END DO
577 
578  END IF
579 !_______________________________________________________________________
580 !
581 ! Set group-to-group scattering moments. Allowed up 4 moments. For
582 ! material 1, start with the base above. Divide by 10 to get next
583 ! moment's data. Repeat that by 2 continuously, up to nmom or 4.
584 !_______________________________________________________________________
585 
586  IF ( nmom > 1 ) THEN
587  slgg(1,2,:,:) = 0.1_r_knd*slgg(1,1,:,:)
588  DO n = 3, nmom
589  slgg(1,n,:,:) = half*slgg(1,n-1,:,:)
590  END DO
591  END IF
592 !_______________________________________________________________________
593 !
594 ! Similar procedure for material 2, but first multiply by 0.8. Then
595 ! reduce the data magnitudes by 0.6 for each successive moment.
596 !_______________________________________________________________________
597 
598  IF ( nmat==2 .AND. nmom>1 ) THEN
599  slgg(2,2,:,:) = 0.8_r_knd*slgg(2,1,:,:)
600  DO n = 3, nmom
601  slgg(2,n,:,:) = 0.6_r_knd*slgg(2,n-1,:,:)
602  END DO
603  END IF
604 !_______________________________________________________________________
605 !_______________________________________________________________________
606 
607  END SUBROUTINE setup_data
